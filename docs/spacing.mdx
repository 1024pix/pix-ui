  `--pix-spacing-1x`
  `--pix-spacing-2x`
  `--pix-spacing-3x`
  `--pix-spacing-4x`
  `--pix-spacing-6x`
  `--pix-spacing-8x`
  `--pix-spacing-10x`
  `--pix-spacing-12x`
